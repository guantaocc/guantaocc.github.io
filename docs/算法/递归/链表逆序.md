## 单链表逆序

```js
// 单链表逆序
class ListNode {
  constructor(value){
    this.value = value
    this.next = null
  }
}

const head = new ListNode(0)
head.next = new ListNode(1)
head.next.next = new ListNode(2)

function reverseList(head){
  if(!head){
    return
  }
  let cur = null
  while(head){
    head.next = cur
    cur = head
    head = head.next
  }
  // 头结点调换
  return cur
}
reverseList(head)
console.log(head.value)
```
