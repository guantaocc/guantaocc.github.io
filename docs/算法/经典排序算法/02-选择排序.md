## 选择排序算法

每一轮选择一个数跟当前数比较
每轮排序结束后 最大或最小的数会放在当前位置上


```js
// 交换
function swap(arr, i, j) {
  let temp
  temp = arr[i]
  arr[i] = arr[j]
  arr[j] = temp
}

function selectSort(arr) {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] < arr[i]) {
        swap(arr, i, j)
      }
    }
    // 每轮结束,最小的数会放在 i 位置上
  }
}

let arr = [4, 8, 6, 9, 4, 3]
// [3, 8, 6, 9, 4, 4]
// [3, 4, 8, 9, 6, 4]
// [3, 4, 4, 9, 8, 6]
// [3, 4, 4, 6, 9, 8]
// [3, 4, 4, 6, 8, 9]
selectSort(arr)
console.log(arr)
```
