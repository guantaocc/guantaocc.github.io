## 冒泡排序算法

将最大或最小的数冒泡 合适的位置之上

时间复杂度 o(n2)
空间复杂度o(n)

```js
// 交换
function swap(arr, i, j) {
  let temp
  temp = arr[i]
  arr[i] = arr[j]
  arr[j] = temp
}

function bubbleSort(arr) {
  let len = arr.length
  for (let i = 0; i < len - 1; i++) {
    for (let j = 0; j < len - i - 1; j++) {
      if (arr[j] > arr[j + 1]) {
        swap(arr, j, j + 1)
      }
    }
    // 每一轮排序最大的数 会冒泡到数组的尾部
    // 第二轮开始不需要再比较此位置
  }
}


let arr = [4, 8, 6, 9, 4, 3]
// [4, 6, 8, 4, 3, 9]
// [4, 6, 8, 4, 3, 9]
// [4, 6, 4, 3, 8, 9]
// [4, 4, 3, 6, 8, 9]
// [4, 3, 4, 6, 8, 9]
// [3, 4, 4, 6, 8, 9]
bubbleSort(arr)
console.log(arr)
```
